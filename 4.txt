
# 디렉토리 무시 예시
build/       # 'build' 디렉토리와 그 하위 모든 파일 무시
tmp/         # 'tmp' 디렉토리와 그 하위 모든 파일 무시

# 특정 파일 무시 예시
*.log        # 모든 .log 파일 무시
config.ini   # config.ini 파일 무시

# 특정 확장자 무시 예시
*.exe        # 모든 .exe 실행 파일 무시
*.dll        # 모든 .dll 라이브러리 파일 무시

# 특정 패턴으로 시작하는 파일 무시 예시
temp-*       # 'temp-'로 시작하는 모든 파일 무시

# 특정 패턴으로 끝나는 파일 무시 예시
*.tmp        # .tmp로 끝나는 모든 파일 무시

# 특정 패턴이 포함된 파일 무시 예시
debug.log    # 'debug.log' 파일 무시

# 경로를 지정하여 파일 무시 예시
doc/*.pdf    # 'doc' 디렉토리에 있는 모든 .pdf 파일 무시

# 디렉토리의 내용물 무시 예시
!src/       # 'src' 디렉토리와 그 내용물은 무시하지 않음
